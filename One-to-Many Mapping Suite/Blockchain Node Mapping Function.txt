package mapper

import (
	"fmt"
)

var NodeMap = map[string]string{
	"Architecture": "peer0.org1.example.com",
	"Structure":    "peer0.org2.example.com",
}

// 通过部门映射到 Fabric 节点
func MapUserToNode(dept string) (string, error) {
	node, ok := NodeMap[dept]
	if !ok {
		return "", fmt.Errorf("no blockchain node for dept=%s", dept)
	}
	return node, nil
}
